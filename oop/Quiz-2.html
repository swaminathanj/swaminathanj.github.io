// U.java
public class U {
	private int x;
	public void set(int xi) { x = xi; }
	public int get() { return x; }
}

// UTest.java
public class UTest {
  public static void main(String[] args) {
    U[] u = new U[10];
    for (int i=0; i<u.length; i++)
      u.set(i);
    System.out.println( u[10].get() );
  }
}
