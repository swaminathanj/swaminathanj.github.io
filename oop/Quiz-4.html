Abstract
class		 
// X.java
public abstract class X {
  protected int x;
  public x() { x = 1; }
  public abstract void set(int x);
  public void print() {
    System.out.println(x);
  }
}

// XX.java
public class XX extends X {
  public void set(int x) {
    super.x = x; // OR this.x = x;
  }
} 

// XTest.java
public class XTest {
  public static void main(String[] args) {
    X x = new XX();
    x.print();
    x.set(5);
    x.print();
  }
}
