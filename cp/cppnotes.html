basic_ops.cpp 
#include <iostream>
using namespace std;

int main() {
  int a, b;
  int sum, difference, product, quotient, reminder;

  cin >> a >> b;

  sum = a+b;
  difference = a-b;
  product = a*b;
  quotient = a/b;
  remainder = a%b;

  cout << "sum = " << sum << endl;
  cout << "difference = " << difference << endl;
  cout << "product = " << product << endl;
  cout << "quotient = " << quotient << endl;
  cout << "remainder = " << remainder << endl;

  return 0;
}

ifelse.cpp 
#include <iostream>
using namespace std;

int main() {
  int a, b;

  cin >> a >> b;

  if (a == b)
    cout << a << " is equal to " << b << endl;
  else
    cout << a << " is not equal to " << b << endl;

  return 0;
}
A program that recieves two
integers a and b, checks if
they are the same or not and
and prints appropriately.

To check if a < b: if (a < b)
To check if a ≤ b: if (a <= b)
To check if a > b: if (a > b)
To check if a ≥ b: if (a >= b)
To check if a ≠ b: if (a != b)

while.cpp 
#include <iostream>
using namespace std;

int main() {
  int a, b;

  int i = 0;  // Initially i = 0
  while (i < 5) {
    cin >> a >> b;
    cout << a+b << endl;
    i = i + 1;  // Incremented by one
  }

  return 0;
}

whilen.cpp 
#include <iostream>
using namespace std;

int main() {
  int a, b;

  int i = 0;  // Initially i = 0
  cin >> n;
  while (i < n) {
    cin >> a >> b;
    cout << a+b << endl;
    i++;  // same as i = i + 1;
  }

  return 0;
}

arraysum.cpp 
#include <iostream>
using namespace std;

int main() {
  int a[5]; // can hold 5 integers

  cin >> a[0] >> a[1] >> a[2] >> a[3] >> a[4];
  int i;
  int sum = 0;
  for (i = 0; i < 5; i++) {
    sum = sum + a[i];
  }
  cout << sum << endl;

  return 0;
}
